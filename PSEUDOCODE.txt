


SET LINE_DEC to a string of 50 "=" characters
SET VAT_RATE = 0.12
SET VAT_TIN
SET MIN
SET SERIAL_NUMBER
SET grocery_name = "Aisle Shop Here"
CREATE an empty list named cart_list
LET THE cash to 0


CREATE INVENTORY as a 2D list
    Each aisle contains:
        aisle name,
        list of items
        each item contains:
            item ID
            item name
            price



FUNCTION show_aisle(aisle_data)

    GET aisle name
    GET list of items

    LOOP forever
        DISPLAY aisle header

        FOR each item in aisle
            COMPUTE vat_added_price =
                item price × (1 + VAT_RATE)

            DISPLAY item ID, item name,and VAT_added_price
        END FOR

        DISPLAY "Type 'back' to return to menu or 'cart' to view cart"

        GET user input - choice
        CONVERT input to lowercase
        REMOVE all extra spaces

        IF choice is "back"
            RETURN to previous menu

        ELSE IF choice is "cart"
            CALL view_cart()

        ELSE IF choice is an integer AND within item range i.e., 0 <choice <= number of items

            FOR each item in items

                IF choice matches item ID

                    ASK user  - choice
                        "Buy item? (Y/N)"
			CONVERT input to uppercase
               		REMOVE all extra spaces

                    IF choice is "Y"

                        ASK quantity in integers

                        ADD item name, item price, quantity TO cart_list

                        DISPLAY
                            "Added to cart!"
                    END IF

                END IF

            END FOR

        ELSE
            DISPLAY "Invalid choice."

        END IF

    END LOOP

END FUNCTION
